Set up login account

    >>> portal = layer['portal']
    >>> memberName = 'siteManager'
    >>> portal.portal_membership.addMember(
    ...         memberName,
    ...         memberName,
    ...         ('Member', 'Manager',),
    ...         '',
    ...         {'fullname': 'Site Manager', 'email': memberName+'@dummy.fr',}
    ...         )
    >>> memberName = 'userManager'
    >>> portal.portal_membership.addMember(
    ...         memberName,
    ...         memberName,
    ...         ('Member',),
    ...         '',
    ...         {'fullname': 'Site Member', 'email': memberName+'@dummy.fr',}
    ...         )

Create plomino database and publish it::

    >>> db = portal.mydb
    >>> temp_storage = getattr(db,'temporary_files',None)
    >>> wf_tool = portal.portal_workflow
    >>> wf_tool.doActionFor(db, 'publish')
    >>> db.manage_setLocalRoles('userManager', ['PlominoManager'])

Create the browser object we'll be using::

    >>> browser = Browser(layer['app'])
    >>> browser.handleErrors = False
    >>> transaction.commit()  # enable the browser to see our changes

Open portal::

    >>> portal_url = portal.absolute_url()
    >>> browser.open(portal_url)

Log in with ``Site Manager`` access rights::

    >>> browser.getLink('Log in').click()
    >>> browser.getControl('Login Name').value = 'siteManager'
    >>> browser.getControl('Password').value = 'siteManager'
    >>> browser.getControl('Log in').click()

Test selection values are correctly validated
====

Create a form with selections

    >>> id = db.invokeFactory('PlominoForm', id='selection_form', title='Selection Form')
    >>> id = db.selection_form.invokeFactory('PlominoField',
    ...             id='selection_1',
    ...             title='Selection 1',
    ...             field_type="SELECTION",
    ...             field_mode="EDITABLE")
    >>> id = db.selection_form.invokeFactory('PlominoField',
    ...             id='selection_2',
    ...             title='Selection 2',
    ...             field_type="SELECTION",
    ...             field_mode="EDITABLE")
    >>> db.selection_form.form_layout = """
    ...    <p><label for="selection_1">Selection 1</label><span class="plominoFieldClass">selection_1</span></p>
    ...    <p><label for="selection_2">Selection 2</label><span class="plominoFieldClass">selection_2</span></p>
    ... """
    >>> db.selection_form.selection_1.selectionlist = ['apple','orange']
    >>> db.selection_form.selection_2.selectionlist = ['apple','orange']
    >>> db.selection_form.selection_2.widget = 'MULTISELECT'
    >>> transaction.commit()
    >>> selection_form_url = portal_url +'/mydb' +'/selection_form'

Check valid select inputs are allowed

    >>> browser.open(selection_form_url)
    >>> browser.getControl(name='selection_1').value = ['apple']
    >>> browser.getControl('Save').click()
    >>> browser.contents
    '...<p><label for="selection_1">Selection 1</label>\n    apple\n    \n\n\n</p>...'

Add an invalid select option and check it is caught as invalid
 
    >>> browser.open(selection_form_url)
    >>> from mechanize import Item
    >>> selection = browser.getControl(name='selection_1')
    >>> ret = Item(selection.mech_control, {"contents": "banana", "value": "banana"})
    >>> selection.value = ['banana']
    >>> browser.getControl('Save').click()
    >>> browser.contents
    '...<li id="error_selection_1">Submitted value (banana) does not match available selection options</li>...'

Check valid multi-select inputs are allowed

    >>> browser.open(selection_form_url)
    >>> browser.getControl(name='selection_2').value = ['apple', 'orange']
    >>> browser.getControl('Save').click()
    >>> browser.contents
    '...<p><label for="selection_2">Selection 2</label>\n    apple\n    \n\n    orange\n    \n\n\n</p>...'

Add an invalid multi-select option and check it is caught as invalid

    >>> browser.open(selection_form_url)
